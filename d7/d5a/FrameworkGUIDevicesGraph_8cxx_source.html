42 NodeColor
43  decideColorForNode(const DeviceInfo& info)
44 {
45  NodeColor result;
46  if (info.active == false) {
47  result.normal = PaletteHelpers::RED;
48  result.hovered = PaletteHelpers::SHADED_RED;
49  return result;
50  }
51  switch (info.maxLogLevel) {
52  case LogParsingHelpers::LogLevel::Error:
53  result.normal = PaletteHelpers::SHADED_RED;
54  result.hovered = PaletteHelpers::RED;
55  result.title = PaletteHelpers::RED;
56  result.title_hovered = PaletteHelpers::DARK_RED;
57  break;
58  case LogLevel::Warning:
59  result.normal = PaletteHelpers::SHADED_YELLOW;
60  result.hovered = PaletteHelpers::YELLOW;
61  result.title = PaletteHelpers::YELLOW;
62  result.title_hovered = PaletteHelpers::DARK_YELLOW;
63  break;
64  case LogLevel::Info:
65  result.normal = PaletteHelpers::SHADED_GREEN;
66  result.hovered = PaletteHelpers::GREEN;
67  result.title = PaletteHelpers::GREEN;
68  result.title_hovered = PaletteHelpers::DARK_GREEN;
69  break;
70  default:
71  result.normal = PaletteHelpers::GRAY;
72  result.hovered = PaletteHelpers::LIGHT_GRAY;
73  result.title = PaletteHelpers::GRAY;
74  result.title_hovered = PaletteHelpers::BLACK;
75  break;
76  }
77  return result;
78 }

104  Node(int id, const char* name, const ImVec2& pos, float value, const ImVec4& color, int inputs_count, int outputs_count)
105  {
106  ID = id;
107  strncpy(Name, name, 31);

116  ImVec2 GetInputSlotPos(int slot_no) const { return ImVec2(Pos.x, Pos.y + Size.y * ((float)slot_no + 1) / ((float)InputsCount + 1)); }
117  ImVec2 GetOutputSlotPos(int slot_no) const { return ImVec2(Pos.x + Size.x, Pos.y + Size.y * ((float)slot_no + 1) / ((float)OutputsCount + 1)); }

23 // a structure describing EventPart
24 // (an elementary constituent of a collision)
25 struct EventPart {
26  EventPart() = default;
27  EventPart(int s, int e) : sourceID(s), entryID(e) {}
28  int sourceID = 0; // the ID of the source (0->backGround; > 1 signal source)
29  // the sourceID should correspond to the chain ID
30  int entryID = 0; // the event/entry ID inside the chain corresponding to sourceID
31 
32  static bool isSignal(EventPart e) { return e.sourceID > 1; }
33  static bool isBackGround(EventPart e) { return !isSignal(e); }

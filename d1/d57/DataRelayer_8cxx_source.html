74 void DataRelayer::processDanglingInputs(std::vector<ExpirationHandler> const& expirationHandlers,
75  ServiceRegistry& services)
76 {
77  for (size_t ti = 0; ti < mTimeslices.size(); ++ti) {
78  // FIXME: for the moment we need to have at least one data input before we
79  // can invoke the dangling inputs helpers.
80  // This is useful for stuff like conditions or histograms, but not
81  // (yet) for timers. For those we would need to have a mechanism
82  // to create valid timeslices.
83  if (mTimeslices[ti].value == INVALID_TIMESLICE) {
84  continue;
85  }
86  assert(mDistinctRoutesIndex.empty() == false);
87  for (size_t ri = 0; ri < mDistinctRoutesIndex.size(); ++ri) {
88  auto& route = mInputRoutes[mDistinctRoutesIndex[ri]];
89  auto& expirator = expirationHandlers[mDistinctRoutesIndex[ri]];
90  auto timestamp = mTimeslices[ti].value;
91  auto& part = mCache[ti * mDistinctRoutesIndex.size() + ri];
92  if (part.header == nullptr && part.payload == nullptr && expirator.checker && expirator.checker(mTimeslices[ti].value)) {
93  assert(ti * mDistinctRoutesIndex.size() + ri < mCache.size());
94  assert(expirator.handler);
95  expirator.handler(services, part, timestamp);
96  mDirty[ti] = true;
97  assert(part.header != nullptr);
98  assert(part.payload != nullptr);
99  }
100  }
101  }
102 }

107 size_t
108 assignInputSpecId(void *data, std::vector<InputRoute> const &routes) {
109  for (size_t ri = 0, re = routes.size(); ri < re; ++ri) {
110  auto &route = routes[ri];
111  const DataHeader* h = o2::header::get<DataHeader*>(data);
112  if (h == nullptr) {
113  return re;
114  }
115 
116  if (DataSpecUtils::match(route.matcher,
117  h->dataOrigin,

119  h->subSpecification)) {
120  return ri;
121  }
122  }

126 DataRelayer::RelayChoice
127 DataRelayer::relay(std::unique_ptr<FairMQMessage> &&header,
128  std::unique_ptr<FairMQMessage> &&payload) {
129  // STATE HOLDING VARIABLES
130  // This is the class level state of the relaying. If we start supporting
131  // multithreading this will have to be made thread safe before we can invoke
132  // relay concurrently.
133  const auto &inputRoutes = mInputRoutes;
134  std::vector<TimesliceId> &timeslices = mTimeslices;
135  auto &cache = mCache;
136  const auto &readonlyCache = mCache;
137  auto& metrics = mMetrics;
138  auto& dirty = mDirty;
139  auto numInputTypes = mDistinctRoutesIndex.size();
140 
141  // IMPLEMENTATION DETAILS
142  //
143  // This returns the identifier for the given input. We use a separate
144  // function because while it's trivial now, the actual matchmaking will
145  // become more complicated when we will start supporting ranges.
146  auto getInput = [&inputRoutes,&header] () -> int {
147  return assignInputSpecId(header->GetData(), inputRoutes);
148  };
149 
150  // This will check if the input is valid. We hide the details so that
151  // in principle the outer code will work regardless of the actual
152  // implementation.
153  auto isValidInput = [](int inputIdx) {
154  // If this is true, it means the message we got does
155  // not match any of the expected inputs.
156  return inputIdx != INVALID_INPUT;
157  };
158 
159  // This will check if the timeslice is valid. We hide the details so that in
160  // principle outer code will work regardless of the actual implementation.
161  auto isValidTimeslice = [](int64_t timesliceId) -> bool {
162  return timesliceId != INVALID_TIMESLICE;
163  };
164 
165  // The timeslice is embedded in the DataProcessingHeader header of the O2
166  // header stack. This is an extension to the DataHeader, because apparently
167  // we do have data which comes without a timestamp, although I am personally
168  // not sure what that would be.
169  auto getTimeslice = [&header, &timeslices]() -> int64_t {
170  const DataProcessingHeader* dph = o2::header::get<DataProcessingHeader*>(header->GetData());
171  if (dph == nullptr) {
172  return -1;
173  }
174  size_t timesliceId = dph->startTime;
175  assert(timeslices.size());
176  return timesliceId;
177  };
178 
179  // Late arrival means that the incoming timeslice is actually older
180  // then the one hold in the current cache.
181  auto isInputFromObsolete = [&timeslices](int64_t timeslice) {
182  auto &current = timeslices[timeslice % timeslices.size()];
183  return current.value > timeslice;
184  };
185 
186  // A cache line is obsolete if the incoming one has a greater timestamp or
187  // if the slot contains an invalid timeslice.
188  auto isCacheEntryObsoleteFor = [&timeslices](int64_t timeslice){
189  auto &current = timeslices[timeslice % timeslices.size()];
190  return current.value == INVALID_TIMESLICE
191  || (current.value < timeslice);
192  };
193 
194  // We need to prune the cache from the old stuff, if any. Otherwise we
195  // simply store the payload in the cache and we mark relevant bit in the
196  // hence the first if.
197  auto pruneCacheSlotFor = [&cache, &numInputTypes, &timeslices, &metrics](int64_t timeslice) {
198  assert(cache.empty() == false);
199  assert(timeslices.size() * numInputTypes == cache.size());
200  size_t slotIndex = timeslice % timeslices.size();
201  // Prune old stuff from the cache, hopefully deleting it...
202  // We set the current slot to the timeslice value, so that old stuff
203  // will be ignored.
204  assert(numInputTypes * slotIndex < cache.size());
205  timeslices[slotIndex].value = timeslice;
206  for (size_t ai = slotIndex*numInputTypes, ae = ai + numInputTypes; ai != ae ; ++ai) {
207  cache[ai].header.reset(nullptr);
208  cache[ai].payload.reset(nullptr);
209  metrics.send({ 0, std::string("data_relayer/") + std::to_string(ai) });
210  }
211  };
212 
213  // We need to check if the slot for the current input is already taken for
214  // the current timeslice.
215  // This should never happen, however given this is dependent on the input
216  // we want to protect again malicious / bad upstream source.
217  auto hasCacheInputAlreadyFor = [&cache, &timeslices, &numInputTypes](int64_t timeslice, int input) {
218  size_t slotIndex = timeslice % timeslices.size();
219  PartRef &currentPart = cache[numInputTypes*slotIndex + input];
220  return (currentPart.payload != nullptr) || (currentPart.header != nullptr);
221  };
222 
223  // Actually save the header / payload in the slot
224  auto saveInSlot = [&header, &payload, &cache, &dirty, &timeslices, &numInputTypes, &metrics](int64_t timeslice, int input) {
225  size_t slotIndex = timeslice % timeslices.size();
226  dirty[slotIndex] = true;
227  auto cacheIdx = numInputTypes * slotIndex + input;
228  PartRef& currentPart = cache[cacheIdx];
229  metrics.send({ 1, std::string("data_relayer/") + std::to_string(cacheIdx) });
230  PartRef ref{std::move(header), std::move(payload)};
231  currentPart = std::move(ref);
232  timeslices[slotIndex] = {timeslice};
233  assert(header.get() == nullptr && payload.get() == nullptr);
234  };
235 
236  // OUTER LOOP
237  //
238  // This is the actual outer loop processing input as part of a given
239  // timeslice. All the other implementation details are hidden by the lambdas
240  auto input = getInput();
241 
242  if (isValidInput(input) == false) {
243  LOG(ERROR) << "A malformed message just arrived";
244  return WillNotRelay;
245  }
246 
247  auto timeslice = getTimeslice();
248  LOG(DEBUG) << "Received timeslice" << timeslice;
249  if (isValidTimeslice(timeslice) == false) {
250  LOG(ERROR) << "Could not determine the timeslice for input";
251  return WillNotRelay;
252  }
253 
254  if (isInputFromObsolete(timeslice)) {
255  LOG(ERROR) << "An entry for timeslice " << timeslice << " just arrived but too late to be processed";
256  return WillNotRelay;
257  }
258 
259  if (isCacheEntryObsoleteFor(timeslice)) {
260  pruneCacheSlotFor(timeslice);
261  }
262  if (hasCacheInputAlreadyFor(timeslice, input)) {
263  LOG(ERROR) << "Got a message with the same header and timeslice twice!!";
264  return WillNotRelay;
265  }
266  saveInSlot(timeslice, input);
267  return WillRelay;
268 }

271 std::vector<DataRelayer::RecordAction>
272 DataRelayer::getReadyToProcess() {
273  // THE STATE
274  std::vector<RecordAction> completed;
275  const auto &cache = mCache;
276  const auto numInputTypes = mDistinctRoutesIndex.size();
277  //
278  // THE IMPLEMENTATION DETAILS
279  //
280  // We use this to bail out early from the check as soon as we find something
281  // which we know is not complete.
282  auto getPartialRecord = [&cache, &numInputTypes](int li) -> gsl::span<const PartRef> {
283  auto offset = li * numInputTypes;
284  assert(cache.size() >= offset + numInputTypes);
285  auto const start = cache.data() + offset;
286  auto const end = cache.data() + offset + numInputTypes;
287  return gsl::span<const PartRef>(start, end);
288  };
289 
290  // These two are trivial, but in principle the whole loop could be parallelised
291  // or vectorised so "completed" could be a thread local variable which needs
292  // merging at the end.
293  auto updateCompletionResults = [&completed](size_t li, CompletionPolicy::CompletionOp op) {
294  completed.push_back({li, op});
295  };
296 
297  auto completionResults = [&completed]() -> std::vector<RecordAction> {
298  return completed;
299  };
300 
301  // THE OUTER LOOP
302  //
303  // To determine if a line is complete, we iterate on all the arguments
304  // and check if they are ready. We do it this way, because in the end
305  // the number of inputs is going to be small and having a more complex
306  // structure will probably result in a larger footprint in any case.
307  // Also notice that ai == inputsNumber only when we reach the end of the
308  // iteration, that means we have found all the required bits.
309  assert(numInputTypes != 0);
310  size_t cacheLines = cache.size() / numInputTypes;
311  assert(cacheLines * numInputTypes == cache.size());
312 
313  for (size_t li = 0; li < cacheLines; ++li) {
314  // We only check the cachelines which have been updated by an incoming
315  // message.
316  if (mDirty[li] == false) {
317  continue;
318  }
319  auto partial = getPartialRecord(li);
320  auto action = mCompletionPolicy.callback(partial);
321  switch (action) {
322  case CompletionPolicy::CompletionOp::Consume:
323  case CompletionPolicy::CompletionOp::Process:
324  case CompletionPolicy::CompletionOp::Discard:
325  updateCompletionResults(li, action);
326  break;
327  case CompletionPolicy::CompletionOp::Wait:
328  break;
329  }
330  // Given we have created an action for this cacheline, we need to wait for
331  // a new message before we look again into the given cacheline.
332  mDirty[li] = false;
333  }
334  return completionResults();
335 }

337 std::vector<std::unique_ptr<FairMQMessage>>
338 DataRelayer::getInputsForTimeslice(size_t timeslice) {
339  const auto numInputTypes = mDistinctRoutesIndex.size();
340  // State of the computation
341  std::vector<std::unique_ptr<FairMQMessage>> messages;
342  messages.reserve(numInputTypes*2);
343  auto &cache = mCache;
344  auto &timeslices = mTimeslices;
345  auto& metrics = mMetrics;
346 
347  // Nothing to see here, this is just to make the outer loop more understandable.
348  auto jumpToCacheEntryAssociatedWith = [](size_t) {
349  return;
350  };
351 
352  // We move ownership so that the cache can be reused once the computation is
353  // finished. We bump by one the timeslice for the given cache entry, so that
354  // in case we get (for whatever reason) an old input, it will be
355  // automatically discarded by the relay method.
356  auto moveHeaderPayloadToOutput = [&messages, &cache, &timeslices, &numInputTypes, &metrics](size_t ti, size_t arg) {
357  auto cacheId = ti * numInputTypes + arg;
358  metrics.send({ 2, "data_relayer/" + std::to_string(cacheId) });
359  messages.emplace_back(std::move(cache[cacheId].header));
360  messages.emplace_back(std::move(cache[cacheId].payload));
361  timeslices[ti % timeslices.size()].value += 1;
362  };
363 
364  // An invalid set of arguments is a set of arguments associated to an invalid
365  // timeslice, so I can simply do that. I keep the assertion there because in principle
366  // we should have dispatched the timeslice already!
367  // FIXME: what happens when we have enough timeslices to hit the invalid one?
368  auto invalidateCacheFor = [&numInputTypes, &timeslices, &cache](size_t ti) {
369  for (size_t ai = ti*numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
370  assert(cache[ai].header.get() == nullptr);
371  assert(cache[ai].payload.get() == nullptr);
372  }
373  timeslices[ti % timeslices.size()] = INVALID_TIMESLICE_ID;
374  };
375 
376  // Outer loop here.
377  jumpToCacheEntryAssociatedWith(timeslice);
378  for (size_t ai = 0, ae = numInputTypes; ai != ae; ++ai) {
379  moveHeaderPayloadToOutput(timeslice, ai);
380  }
381  invalidateCacheFor(timeslice);
382 
383  return std::move(messages);
384 }

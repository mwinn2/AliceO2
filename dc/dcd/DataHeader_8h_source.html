178 template <typename T, std::size_t N, std::size_t pos = 0, bool suppressAssert = false>
179 constexpr T String2(const char (&str)[N])
180 {
181  static_assert(std::is_integral<T>::value, "Non integral types not compatible with String2<type>");
182  static_assert(N >= pos, "Position is larger than the length of string");
183  static_assert(suppressAssert || N - pos <= sizeof(T) + 1,
184  "String parameter is longer than the size of the data type"
185  );
186  return((T) str[0+pos] |
187  (str[0+pos] && sizeof(T) >= 2 ? ((T) str[1+pos] << (sizeof(T) >= 2 ? 8 : 0) |
188  (str[1+pos] && sizeof(T) >= 4 ? ((T) str[2+pos] << (sizeof(T) >= 4 ? 16 : 0) |
189  (str[2+pos] && sizeof(T) >= 4 ? ((T) str[3+pos] << (sizeof(T) >= 4 ? 24 : 0) |
190  (str[3+pos] && sizeof(T) >= 8 ? ((T) str[4+pos] << (sizeof(T) >= 8 ? 32 : 0) |
191  (str[4+pos] && sizeof(T) >= 8 ? ((T) str[5+pos] << (sizeof(T) >= 8 ? 40 : 0) |
192  (str[5+pos] && sizeof(T) >= 8 ? ((T) str[6+pos] << (sizeof(T) >= 8 ? 48 : 0) |
193  (str[6+pos] && sizeof(T) >= 8 ? ((T) str[7+pos] << (sizeof(T) >= 8 ? 56 : 0) )

278  void runtimeInit(const char* string, short length = -1) {
279  char* target = str;
280  char* targetEnd = target;
281  if (length >= 0 && length < (int)N) targetEnd += length;
282  else targetEnd += N;
283  const char* source = string;
284  for ( ; source != nullptr && target < targetEnd && *source !=0; ++target, ++source) *target = *source;
285  targetEnd = str + N;
286  for ( ; target < targetEnd; ++target) *target = 0;
287  // require the string to be not longer than the descriptor size
288  if (source != nullptr && (*source == 0 || (length >= 0 && length <= (int)N))) {
289  } else {
290  throw std::invalid_argument("argument must not be longer than descriptor size");
291  }
292  }

447 template <typename HeaderType, typename std::enable_if_t<std::is_pointer<HeaderType>::value, int> = 0>
448 auto get(const o2::byte* buffer, size_t /*len*/ = 0)
449 {
450  using HeaderConstPtrType = const typename std::remove_pointer<HeaderType>::type*;
451  using HeaderValueType = typename std::remove_pointer<HeaderType>::type;
452 
453  const BaseHeader* current = BaseHeader::get(buffer);
454  if (!current)
455  return HeaderConstPtrType{ nullptr };
456  if (current->description == HeaderValueType::sHeaderType)
457  return reinterpret_cast<HeaderConstPtrType>(current);
458  while ((current = current->next())) {
459  if (current->description == HeaderValueType::sHeaderType)
460  return reinterpret_cast<HeaderConstPtrType>(current);
461  }
462  return HeaderConstPtrType{ nullptr };

579  template <typename T, typename... Args>
580  static o2::byte* inject(o2::byte* here, T&& h, Args&&... args) noexcept
581  {
582  auto alsohere = inject(here, h);
583  // the type might be a stack itself, loop through headers and set the flag in the last one
584  if (h.size() > 0) {
585  BaseHeader* next = BaseHeader::get(here);
586  while (next->flagsNextHeader) {
587  next = next->next();
588  }
589  next->flagsNextHeader = hasNonEmptyArg(args...);
590  }
591  return inject(alsohere, args...);
592  }
